\newcommand{\PreEffLength}{5em} % both pre & effs (same length)
\newcommand{\ActionWidth}{6em}  % width of actions
\newcommand{\ActionHeight}{5em} % height of actions

% Make sure that scheme definitions don't have empty lines; this will not compile!

\scheme{pickup}{2}{
  text = {\textbf{pickup}\\(#1,#2)},
  pres = {
    {gF(#1)},
    {onT(#2)},
    {clr(#2)}
  },
  effs = {
    {{$\neg$gF(#1)}},
    {gH(#1,#2)},
    {{$\neg$clr(#2)}},
    {{$\neg$onT(#2)}}
  },
  limb length = \PreEffLength,
  height      = \ActionHeight,
  width       = \ActionWidth
}
